trait ConsoleLogger extends Logger{
  override def logger{ //因为Logger已经实现了，所以要加override，否则不用加
    println("ConsoleLogger")
  }
}

class MyTest extends Logger{
}

object Test extends App{
  val c = new MyTest with ConsoleLogger  //物质混入对象
  c.logger  //调用ConsoleLogger
  val x = new MyTest with Logger 
  x.logger  //调用ConsoleLogger，不会用Logger的，这里应该也会用子特质的
  val d = new MyTest   
  d.logger   //调用Logger
}

如果类混入了某个特质X，有方法A，则对象混入另一个拥有方法A的特质Y时会报错，因为是方法冲突，除非X和Y有继承关系，或者X ext Y，或者Y ext X

trait Logger{

class MyTest extends Logger{
}

object Test extends App{
